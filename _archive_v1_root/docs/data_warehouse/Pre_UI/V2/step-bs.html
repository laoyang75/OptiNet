detect_collision.sql

版本
S-005
步骤

用途

WITH cell_clusters AS ( SELECT bs_id, lac, cell_id, ST_ClusterDBSCAN( ST_SetSRID(ST_MakePoint(lon, lat), 4326), eps := :dbscan_eps, minpoints := :dbscan_minpoints ) OVER ( PARTITION BY bs_id, lac ) AS cluster_id FROM dim_cell_stats WHERE is_trusted = true ), cluster_centers AS ( SELECT bs_id, lac, cluster_id, ST_Centroid(ST_Collect(geom)) AS center, count(*) AS cell_count FROM cell_clusters GROUP BY bs_id, lac, cluster_id ), cluster_distances AS ( SELECT a.bs_id, a.lac, ST_Distance(a.center::geography, b.center::geography) / 1000.0 AS dist_km FROM cluster_centers a JOIN cluster_centers b ON a.bs_id = b.bs_id AND a.cluster_id < b.cluster_id ) SELECT bs_id, lac, max(dist_km) AS max_cluster_dist, count(DISTINCT cluster_id) AS cluster_count FROM cluster_distances WHERE max(dist_km) > :collision_threshold GROUP BY bs_id, lac;
